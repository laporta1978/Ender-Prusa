  G1 Z20 F900      # Raise Z away from print
  G90
  G1 X245 Y218.5 F5000 # Move to purge area
  G91
  G1 E-450 F600 # Retract 450mm of filament
  G90

  G4 P10000 # Pause for 10 seconds
  G1 E150 F400 # Prime the nozzle
